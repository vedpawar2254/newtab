      ```typescript
      function handleDragOver(event: DragOverEvent) {
        const { active, over } = event;
        if (!over) return;

        const activeId = active.id as string;
        const overId = over.id as string;

        // Find which column each item belongs to
        const activeColumn = findColumnForTask(activeId);
        const overColumn = findColumnForTask(overId) ?? overId; // overId might be a column ID

        if (!activeColumn || !overColumn || activeColumn === overColumn) return;

        // Move task to new column in state (optimistic update)
        const overTasks = tasks.filter(t => t.columnId === overColumn);
        const overIndex = overTasks.findIndex(t => t.id === overId);
        const newIndex = overIndex >= 0 ? overIndex : overTasks.length;

        // Update local state: change the task's columnId
        // This is an optimistic local update -- persist happens in onDragEnd
        useTaskStore.getState().reorderTask(activeId, overColumn as string, newIndex);
      }
      ```
    - **onDragEnd**:
      ```typescript
      function handleDragEnd(event: DragEndEvent) {
        setActiveTaskId(null);
        const { active, over } = event;
        if (!over || active.id === over.id) return;

        const activeId = active.id as string;
        const overId = over.id as string;
        const activeColumn = findColumnForTask(activeId);
        const overColumn = findColumnForTask(overId) ?? overId;

        if (activeColumn === overColumn) {
          // Reorder within same column
          const columnTasks = tasks
            .filter(t => t.columnId === activeColumn)
            .sort((a, b) => a.order - b.order);
          const oldIndex = columnTasks.findIndex(t => t.id === activeId);
          const newIndex = columnTasks.findIndex(t => t.id === overId);
          if (oldIndex !== newIndex) {
            useTaskStore.getState().reorderTask(activeId, activeColumn!, newIndex);
          }
        }
        // Cross-column already handled in onDragOver
      }
      ```
